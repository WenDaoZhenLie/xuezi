## **类(class)**
```python
class ClassName():
    """
    类命名：大驼峰
    类属性和类方法被所有类实例所共享
    """

    # 类属性
    class_attribute = None

    # 类方法
    @classmethod
    def class_method(cls):
        """
        类方法的第一个参数表示类本身，由解释器自动绑定
        """
        pass

    # 静态方法
    @staticmethod
    def static_method(*args, **kwargs):
        pass

    # 构造方法
    def __init__(self, public_attribute):
        """
        对象方法的第一个参数表示当前对象，由解释器自动绑定
        """
        # 对象属性
        self.public_attribute = public_attribute

    # 对象方法
    def public_method(self, *parameters):
        pass

    # 抽象方法
    # instancemethod/classmethod/staticmethod
    @abstractmethod
    def atstract_method():
        pass
```

## **封装**
```python
class ClassName():
    """
    __：私有
    """
    def __init__(self, private_attribute):
        # 对象私有属性
        self.__private_attribute = private_attribute

    # 对象私有方法
    def __private_method(self,*parameters):
        pass

    # obj.private_attribute
    @property
    def private_attribute(self):
        return self.__private_attribute

    # obj.private_attribute = None
    @private_attribute.setter
    def private_attribute(self, private_attribute):
        self.__private_attribute = private_attribute
```

## **元类**
```python
"""
默认元类
"""
class type(object):
    pass

"""
自定义元类
"""
class className(type):
    @classmethod
    def __prepare__(mcs, name, bases):
        return super().__prepare__(name, bases)


    @staticmethod
    def __new__(mcs, name, bases, dict):
        return super().__new__(mcs, name, bases, dict)


    def __init__(self, name, bases, dict):
        super().__init__(name, bases, dict)


    def __call__(self, *args, **kwargs):
        pass
```
